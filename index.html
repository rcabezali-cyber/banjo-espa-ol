    <script>

        const apiKey = ""; // API Key se inyecta automáticamente

 

        async function callGemini(prompt, systemPrompt = "Eres un experto profesor de banjo de 5 cuerdas.") {

            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;

           

            const payload = {

                contents: [{ parts: [{ text: prompt }] }],

                systemInstruction: { parts: [{ text: systemPrompt }] }

            };

 

            for (let i = 0; i < 5; i++) {

                try {

                    const response = await fetch(url, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });

                    const data = await response.json();

                    return data.candidates?.[0]?.content?.parts?.[0]?.text;

                } catch (e) {

                    await new Promise(r => setTimeout(r, Math.pow(2, i) * 1000));

                }

            }

            return "Lo siento, ha habido un error al conectar con el cerebro del banjo. Inténtalo de nuevo.";

        }

 

        // Feature 1: Generador de Rutinas

        async function generateRoutine() {

            const level = document.getElementById('userLevel').value;

            const time = document.getElementById('userTime').value;

            const btn = document.getElementById('routineBtn');

            const spinner = document.getElementById('loadingSpinner');

            const text = document.getElementById('btnText');

            const output = document.getElementById('routineOutput');

            const outputText = document.getElementById('routineText');

 

            btn.disabled = true;

            spinner.classList.remove('hidden');

            text.innerText = "Pensando ejercicios...";

 

            const prompt = `Crea una rutina de práctica de banjo para un nivel ${level} que dure ${time}. Divide el tiempo en secciones: Calentamiento, Técnica/Rolls, y Repertorio. Usa un tono motivador.`;

            const result = await callGemini(prompt, "Eres un profesor de banjo bluegrass apasionado. Devuelve la rutina en formato de lista clara.");

 

            outputText.innerText = result;

            output.classList.remove('hidden');

           

            btn.disabled = false;

            spinner.classList.add('hidden');

            text.innerText = "Crear Rutina Personalizada ✨";

           

            window.scrollTo({ top: output.offsetTop - 100, behavior: 'smooth' });

        }

 

        // Feature 2: Chat Asistente

        function toggleAIChat() {

            const chat = document.getElementById('ai-chat-window');

            chat.classList.toggle('hidden');

        }

 

        async function sendChatMessage() {

            const input = document.getElementById('chat-input');

            const container = document.getElementById('chat-messages');

            const query = input.value.trim();

            if(!query) return;

 

            // User message

            const userMsg = document.createElement('div');

            userMsg.className = "bg-stone-200 text-stone-800 p-3 rounded-2xl rounded-tr-none text-sm self-end ml-8";

            userMsg.innerText = query;

            container.appendChild(userMsg);

            input.value = "";

            container.scrollTop = container.scrollHeight;

 

            // Loading state

            const loadingMsg = document.createElement('div');

            loadingMsg.className = "text-xs text-stone-400 italic";

            loadingMsg.innerText = "IA está afinando sus cuerdas...";

            container.appendChild(loadingMsg);

 

            const response = await callGemini(query, "Eres un asistente experto para el blog 'Banjo en Español'. Responde dudas sobre bluegrass, banjo de 5 cuerdas y teoría musical de forma breve y amable.");

           

            container.removeChild(loadingMsg);

 

            // AI message

            const aiMsg = document.createElement('div');

            aiMsg.className = "bg-wood-100 text-wood-900 p-3 rounded-2xl rounded-tl-none text-sm mr-8";

            aiMsg.innerText = response;

            container.appendChild(aiMsg);

            container.scrollTop = container.scrollHeight;

        }

 

        // Quick Explainer

        async function askGemini(question) {

            toggleAIChat();

            const input = document.getElementById('chat-input');

            input.value = question;

            sendChatMessage();

        }
